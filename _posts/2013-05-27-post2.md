    stateMon<-confMon%>%group_by(state)%>%summarise(count = n())%>% mutate(region=tolower(state))

    states <- map_data('state')

    states_all <- data.frame(region=unique(states$region))

    stateMon <- left_join(states_all,stateMon,by='region')
